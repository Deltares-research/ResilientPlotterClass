# Read unstructured data
da = xr.open_dataarray(file_path_tif).sel(band=1)
ds = da.to_dataset()

# Read structured data
uds = xu.open_dataset(file_path_nc).isel(time=10)
uds.grid.edge_node_connectivity = uds.grid.edge_node_connectivity[uds.grid.validate_edge_node_connectivity()]
uds.grid.set_crs(da.rio.crs) # Set coordinate reference system since .nc file does not contain it
uda = uds['mesh2d_flowelem_bl']

# Read geometries
gdf_aoi = gpd.read_file(file_path_aoi)

# Plot settings
data_type = 'bathymetry'
extent_type = 'aoi'

# Plot structured data
fig, axs = rpc.subplots(4, 2, figsize=(16, 16))
axs = axs.flatten()
rpc.pcolormesh(da, ax=axs[0], data_type=data_type, extent_type=extent_type, skip=10)
rpc.imshow(da,     ax=axs[1], data_type=data_type, extent_type=extent_type, skip=10)
rpc.contourf(da,   ax=axs[2], data_type=data_type, extent_type=extent_type, skip=10)
rpc.contour(da,    ax=axs[3], data_type=data_type, extent_type=extent_type, skip=10)
rpc.scatter(ds,    ax=axs[4], data_type=data_type, extent_type=extent_type, skip=10,  x='x', y='y', hue='band_data')
rpc.quiver(ds,     ax=axs[5], data_type='general', extent_type=extent_type, skip=100, x='x', y='y', u='band_data', v='band_data')
rpc.streamplot(ds, ax=axs[6], data_type='general', extent_type=extent_type, skip=100, x='x', y='y', u='band_data', v='band_data')
rpc.basemap(ax=axs[7], map_type='satellite', extent_type=extent_type)
for ax, label in zip(axs, ['pcolormesh', 'imshow', 'contourf', 'contour', 'scatter', 'quiver', 'streamplot', 'basemap']):
    ax.set_xlabel('')
    ax.set_ylabel('')
    ax.set_title(label)
fig.tight_layout()
